/**
 * ESP32-C3 + MPU6050 Fall Detection (Bản Tích Hợp MQTT MedPulse Cloud)
 * =================================================================
 */

#include <Arduino.h>
#include <Wire.h>
#include <WiFi.h>
#include <PubSubClient.h>
#include "MPU6050.h"

// ─────────────────────────────────────────────────────────────────────────────
// CONFIGURATION (WiFi & MQTT)
// ─────────────────────────────────────────────────────────────────────────────
#define WIFI_SSID         "Wokwi-GUEST"
#define WIFI_PASSWORD     ""
#define MQTT_BROKER       "broker.hivemq.com"
#define MQTT_PORT         1883
#define MQTT_CLIENT_ID    "medpulse-mobile-c3-778899"

// MÃ BỆNH NHÂN DI ĐỘNG: Gán cố định thẻ để khớp với Backend (Ví dụ: RFID-1002)
String mobilePatientId = "RFID-1002"; 

// I2C CONFIG (ESP32-C3 Wokwi Pins)
#define SDA_PIN   8
#define SCL_PIN   9
#define INT_PIN   4
#define ALERT_PIN 10

// ─────────────────────────────────────────────────────────────────────────────
// THRESHOLDS & TIMING
// ─────────────────────────────────────────────────────────────────────────────
#define FALL_ACCEL_LOW_THRESHOLD   0.5f
#define FALL_ACCEL_HIGH_THRESHOLD  1.8f
#define IMPACT_THRESHOLD           2.2f
#define MOTIONLESS_ACCEL_DELTA     0.12f
#define MOTIONLESS_GYRO_DELTA      10.0f

#define FALL_DURATION_MS           80
#define IMPACT_WINDOW_MS           3000
#define MOTIONLESS_DURATION_MS     2000
#define ALERT_DURATION_MS          5000
#define SAMPLE_RATE_MS             10
#define MQTT_RETRY_INTERVAL        5000

// ─────────────────────────────────────────────────────────────────────────────
// STATE MACHINE & OBJECTS
// ─────────────────────────────────────────────────────────────────────────────
enum FallState {
    STATE_IDLE,
    STATE_FALLING,
    STATE_IMPACT,
    STATE_MOTIONLESS,
    STATE_ALERT
};

struct SensorData {
    float ax, ay, az;
    float gx, gy, gz;
    float accelMagnitude;
    unsigned long timestamp;
};

struct MotionStats {
    float accelDelta;
    float gyroDelta;
};

MPU6050 mpu;
WiFiClient espClient;
PubSubClient mqtt(espClient);

SensorData currentData, prevData;
MotionStats motionStats;
FallState currentState = STATE_IDLE;

volatile bool intTriggered = false;
unsigned long stateEntryTime   = 0;
unsigned long fallDetectedTime = 0;
unsigned long impactTime       = 0;
unsigned long motionlessStart  = 0;
unsigned long lastSampleTime   = 0;
unsigned long lastMqttRetry    = 0;
unsigned long lastBatteryTime  = 0;

bool alertActive = false;
int fallCount = 0;

// ─────────────────────────────────────────────────────────────────────────────
// NETWORK FUNCTIONS
// ─────────────────────────────────────────────────────────────────────────────
void connectWifi() {
    Serial.print("[WiFi] Connecting");
    WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
    while (WiFi.status() != WL_CONNECTED) {
        delay(500); Serial.print(".");
    }
    Serial.println("\n[WiFi] Connected: " + WiFi.localIP().toString());
}

bool connectMqtt() {
    if (mqtt.connected()) return true;
    Serial.print("[MQTT] Connecting Device to Cloud...");
    if (mqtt.connect(MQTT_CLIENT_ID)) {
        Serial.println(" OK");
        return true;
    } else {
        Serial.printf(" FAILED (%d)\n", mqtt.state());
        return false;
    }
}

void publishMobileMetric(const char* metric, float value) {
    if (!mqtt.connected()) return;
    String topic = "medpulse/health/" + mobilePatientId + "/" + metric;
    String payload = String(value, 1);
    mqtt.publish(topic.c_str(), payload.c_str(), true);
    Serial.printf("   🚀 [CLOUD] Send -> %s = %s\n", topic.c_str(), payload.c_str());
}

// ─────────────────────────────────────────────────────────────────────────────
// ISR
// ─────────────────────────────────────────────────────────────────────────────
void IRAM_ATTR mpuISR() {
    intTriggered = true;
}

// ─────────────────────────────────────────────────────────────────────────────
// SENSOR HELPERS
// ─────────────────────────────────────────────────────────────────────────────
float calculateMagnitude(float x, float y, float z) {
    return sqrt(x * x + y * y + z * z);
}

void readSensor() {
    int16_t ax_raw, ay_raw, az_raw;
    int16_t gx_raw, gy_raw, gz_raw;

    mpu.getMotion6(&ax_raw, &ay_raw, &az_raw, &gx_raw, &gy_raw, &gz_raw);

    currentData.ax = ax_raw / 8192.0f;
    currentData.ay = ay_raw / 8192.0f;
    currentData.az = az_raw / 8192.0f;

    currentData.gx = gx_raw / 65.5f;
    currentData.gy = gy_raw / 65.5f;
    currentData.gz = gz_raw / 65.5f;

    currentData.accelMagnitude = calculateMagnitude(currentData.ax, currentData.ay, currentData.az);
    currentData.timestamp = millis();
}

// ─────────────────────────────────────────────────────────────────────────────
// DETECTION ALGORITHMS
// ─────────────────────────────────────────────────────────────────────────────
bool detectFall() {
    float mag = currentData.accelMagnitude;
    if (mag < FALL_ACCEL_LOW_THRESHOLD) return true;
    if (mag > FALL_ACCEL_HIGH_THRESHOLD && motionStats.gyroDelta > 40) return true;
    return false;
}

bool detectImpact() {
    return (currentData.accelMagnitude > IMPACT_THRESHOLD);
}

bool detectMotionless() {
    return (motionStats.accelDelta < MOTIONLESS_ACCEL_DELTA && motionStats.gyroDelta < MOTIONLESS_GYRO_DELTA);
}

bool detectRecovery() {
    return (currentData.accelMagnitude > 0.8f && currentData.accelMagnitude < 1.3f && motionStats.gyroDelta > 40);
}

// ─────────────────────────────────────────────────────────────────────────────
// STATE MACHINE (Đã tích hợp truyền thông đám mây)
// ─────────────────────────────────────────────────────────────────────────────
void stateMachine() {
    unsigned long now = millis();

    switch (currentState) {
        case STATE_IDLE:
            if (detectFall()) {
                currentState = STATE_FALLING;
                stateEntryTime = now;
                fallDetectedTime = now;
                Serial.println("\n[PHASE 1] FALL DETECTED (Mất trọng lực)");
            }
            break;

        case STATE_FALLING:
            if (now - stateEntryTime >= FALL_DURATION_MS) {
                currentState = STATE_IMPACT;
                stateEntryTime = now;
                Serial.println("[PHASE 2] WAITING IMPACT (Đang rơi tự do)");
            }
            if (currentData.accelMagnitude > 0.8f && currentData.accelMagnitude < 1.3f && motionStats.gyroDelta < 20) {
                Serial.println("[PHASE 1] FALSE ALARM (Nhiễu tín hiệu)");
                currentState = STATE_IDLE;
            }
            break;

        case STATE_IMPACT:
            if (detectImpact()) {
                currentState = STATE_MOTIONLESS;
                impactTime = now;
                motionlessStart = now;
                stateEntryTime = now;
                Serial.println("[PHASE 3] IMPACT DETECTED (Va chạm mạnh)");
            } else if (now - stateEntryTime > IMPACT_WINDOW_MS) {
                Serial.println("[PHASE 2] NO IMPACT TIMEOUT");
                currentState = STATE_IDLE;
            }
            break;

        case STATE_MOTIONLESS:
            if (detectMotionless()) {
                if (now - motionlessStart >= MOTIONLESS_DURATION_MS) {
                    currentState = STATE_ALERT;
                    
                    // KÍCH HOẠT CẢNH BÁO TẠI CHỖ & GỬI LÊN CLOUD
                    fallCount++;
                    alertActive = true;
                    digitalWrite(ALERT_PIN, HIGH);
                    
                    Serial.println("\n=================================");
                    Serial.printf(" 🚨 TÉ NGÃ ĐÃ XÁC NHẬN (#%d)\n", fallCount);
                    Serial.println("=================================\n");

                    // Bắn tín hiệu khẩn cấp lên Broker
                    publishMobileMetric("fall", 1.0); 
                    publishMobileMetric("safe", 0.0); // Ngã coi như rơi vào vùng nguy hiểm
                }
            } else {
                motionlessStart = now;
                if (detectRecovery()) {
                    Serial.println("[RECOVERY] Người bệnh tự ngồi dậy trước khi báo động");
                    currentState = STATE_IDLE;
                }
            }
            if (now - stateEntryTime > 8000) {
                Serial.println("[PHASE 3] MOTIONLESS TIMEOUT");
                currentState = STATE_IDLE;
            }
            break;

        case STATE_ALERT:
            // TỰ HỒI PHỤC: Người bệnh tự đứng dậy hoặc nhân viên y tế nâng đỡ
            if (detectRecovery()) {
                Serial.println("\n[RECOVERY] PHÁT HIỆN NGƯỜI BỆNH ĐÃ ĐỨNG DẬY!");
                digitalWrite(ALERT_PIN, LOW);
                alertActive = false;
                
                // Gửi bản tin giải phóng trạng thái nguy hiểm về Backend
                publishMobileMetric("fall", 0.0);
                publishMobileMetric("safe", 1.0);
                
                currentState = STATE_IDLE;
                stateEntryTime = 0;
            }
            break;
    }
}

// ─────────────────────────────────────────────────────────────────────────────
// SETUP
// ─────────────────────────────────────────────────────────────────────────────
void setup() {
    Serial.begin(115200);
    delay(500);
    Serial.println("\n═════════════════════════════════════");
    Serial.println("  MedPulse — THIẾT BỊ DI ĐỘNG C3  ");
    Serial.println("═════════════════════════════════════");

    pinMode(ALERT_PIN, OUTPUT);
    digitalWrite(ALERT_PIN, LOW);
    pinMode(INT_PIN, INPUT);
    attachInterrupt(digitalPinToInterrupt(INT_PIN), mpuISR, RISING);

    Wire.begin(SDA_PIN, SCL_PIN);
    Wire.setClock(400000);

    Serial.print("[MPU6050] Connecting...");
    mpu.initialize();
    if (!mpu.testConnection()) {
        Serial.println(" FAILED!");
        while (1) {
            digitalWrite(ALERT_PIN, !digitalRead(ALERT_PIN));
            delay(200);
        }
    }
    Serial.println(" OK");

    mpu.setFullScaleAccelRange(MPU6050_ACCEL_FS_4);
    mpu.setFullScaleGyroRange(MPU6050_GYRO_FS_500);
    mpu.setDLPFMode(MPU6050_DLPF_BW_42);
    mpu.setIntDataReadyEnabled(true);

    // KẾT NỐI MẠNG CLOUD
    connectWifi();
    mqtt.setServer(MQTT_BROKER, MQTT_PORT);
    connectMqtt();

    // Khởi tạo trạng thái ban đầu cho máy chủ sạch lỗi
    publishMobileMetric("fall", 0.0);
    publishMobileMetric("safe", 1.0);

    readSensor();
    prevData = currentData;
    Serial.println("\n[SYSTEM] Mobile Device Sẵn Sàng Đeo Trên Người.");
}

// ─────────────────────────────────────────────────────────────────────────────
// MAIN LOOP
// ─────────────────────────────────────────────────────────────────────────────
void loop() {
    unsigned long now = millis();

    // Quản lý kết nối MQTT không chặn để bảo vệ luồng đọc thời gian thực
    if (!mqtt.connected()) {
        if (now - lastMqttRetry >= MQTT_RETRY_INTERVAL) {
            lastMqttRetry = now;
            connectMqtt();
        }
    } else {
        mqtt.loop();
    }

    // Đọc cảm biến nhịp 10ms
    bool shouldSample = false;
    if (intTriggered) { shouldSample = true; intTriggered = false; }
    if (now - lastSampleTime >= SAMPLE_RATE_MS) { shouldSample = true; }

    if (shouldSample) {
        lastSampleTime = now;
        readSensor();

        // Tính toán vi sai chuyển động
        motionStats.accelDelta = abs(currentData.accelMagnitude - prevData.accelMagnitude);
        float gyroCurrent = calculateMagnitude(currentData.gx, currentData.gy, currentData.gz);
        float gyroPrev = calculateMagnitude(prevData.gx, prevData.gy, prevData.gz);
        motionStats.gyroDelta = abs(gyroCurrent - gyroPrev);

        // Chạy thuật toán máy trạng thái
        stateMachine();

        prevData = currentData;
    }

    // Định kỳ gửi dung lượng pin thiết bị di động (Mỗi 10 giây gửi 1 lần)
    if (now - lastBatteryTime >= 10000) {
        lastBatteryTime = now;
        float mockBattery = 88.0f + (random(-10, 10) / 10.0f); // Giả lập dung lượng tiêu hao pin
        publishMobileMetric("battery", mockBattery);
    }

    // Khống chế tắt còi tự động nếu rơi vào trạng thái Alert quá lâu không ai cứu
    if (alertActive && (now - impactTime > ALERT_DURATION_MS)) {
        Serial.println("[ALERT] Cảnh báo Timeout tự động hạ còi (Chuyển chế độ chờ cứu hộ)");
        digitalWrite(ALERT_PIN, LOW);
        alertActive = false;
        
        // Trả trạng thái máy về IDLE để sẵn sàng bắt cú ngã tiếp theo nếu có
        currentState = STATE_IDLE; 
    }
}
